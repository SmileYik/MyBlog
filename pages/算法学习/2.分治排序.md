# 分治排序
  
## java 代码实现
  
```
  public static void mergeSort(int[] A, int start, int end) {
    if (start < end - 1) {
      int mid = (start + end) / 2;
      mergeSort(A, start, mid);
      mergeSort(A, mid, end);
      merge(A, start, mid, end);
    }
  }
  
  public static void merge(int[] A, int start, int mid, int end) {
    int[] R = new int[mid - start];
    int[] L = new int[end - mid];
    
    for (int i = 0; i < R.length; ++i) {
      R[i] = A[start + i];
    }
     
    for (int i = 0; i < L.length; ++i) {
      L[i] = A[mid + i];
    }
    
    int indexR = 0;
    int indexL = 0;
    for (int i = start; i < end; ++i) {
      if (indexR >= R.length) {
        A[i] = L[indexL++];
      } else if (indexL >= L.length) {
        A[i] = R[indexR++];
      } else if (R[indexR] > L[indexL]) {
        A[i] = L[indexL++];
      } else {
        A[i] = R[indexR++];
      }
    }
  }
```
